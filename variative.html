1.1
Развёртывание Nextcloud в Docker
BR-SRV — MySQL + Nextcloud, рабочий каталог /opt/nextcloud
Задание Установить Docker и Docker Compose, создать /opt/nextcloud, разработать docker-compose.yml с двумя сервисами: MySQL и Nextcloud. Настроить volumes. Запустить контейнеры.
📦 Пакеты docker-ce, docker-ce-cli, docker-compose уже предустановлены. Docker-образы nextcloud:latest и mysql:8.0 можно скачать или загрузить из tar.

Запуск Docker

# Запускаем Docker и добавляем в автозагрузку
systemctl enable docker --now

# Проверяем статус
systemctl status docker

# Проверяем версию
docker --version
docker-compose --version

Загрузка образов

Способ А — из интернета (если есть доступ):

# Скачиваем образы Nextcloud и MySQL
docker pull nextcloud:latest
docker pull mysql:8.0

# Проверяем список образов
docker image ls

Способ Б — из tar-архивов в /root:

# Если архивы уже скачаны и лежат в /root/
ls /root/
docker load < /root/nextcloud_latest.tar
docker load < /root/mysql_8.0.tar

Создание рабочего каталога

# Создаём рабочий каталог для Nextcloud
mkdir -p /opt/nextcloud

# Создаём поддиректории для volumes
mkdir -p /opt/nextcloud/nextcloud_data
mkdir -p /opt/nextcloud/mysql_data

# Переходим в рабочий каталог
cd /opt/nextcloud

Создание docker-compose.yml

# Создаём файл конфигурации
nano /opt/nextcloud/docker-compose.yml

Содержимое файла:

version: '3.8'

services:

  db:
    image: mysql:8.0
    container_name: nextcloud_db
    restart: always
    volumes:
      - /opt/nextcloud/mysql_data:/var/lib/mysql
    environment:
      MYSQL_ROOT_PASSWORD: P@ssw0rd
      MYSQL_DATABASE: nextcloud
      MYSQL_USER: nextcloud
      MYSQL_PASSWORD: P@ssw0rd
    command: --default-authentication-plugin=mysql_native_password

  app:
    image: nextcloud:latest
    container_name: nextcloud_app
    restart: always
    ports:
      - "8080:80"
    volumes:
      - /opt/nextcloud/nextcloud_data:/var/www/html
    environment:
      MYSQL_HOST: db
      MYSQL_DATABASE: nextcloud
      MYSQL_USER: nextcloud
      MYSQL_PASSWORD: P@ssw0rd
      NEXTCLOUD_ADMIN_USER: admin
      NEXTCLOUD_ADMIN_PASSWORD: P@ssw0rd
    depends_on:
      - db
📌 Пояснение ключевых параметров volumes: /opt/nextcloud/mysql_data:/var/lib/mysql — данные MySQL хранятся на хосте, не теряются при перезапуске контейнера.
volumes: /opt/nextcloud/nextcloud_data:/var/www/html — файлы Nextcloud тоже на хосте.
ports: "8080:80" — внешний порт 8080 → внутренний 80 контейнера.
depends_on: db — app стартует только после db.
MYSQL_HOST: db — имя сервиса db используется как hostname внутри Docker-сети.

cd /opt/nextcloud

# Смотрим статус контейнеров

# Ожидаем: nextcloud_db и nextcloud_app со статусом "Up"

# Если нужно посмотреть логи запуска
docker-compose logs -f

1.2
Самоподписанный SSL-сертификат
BR-SRV — openssl, сертификат для nginx
📦 Пакет openssl-gost-engine уже предустановлен, openssl доступен.

Создание директории для сертификатов

# Создаём папку для хранения SSL-сертификатов nginx
mkdir -p /etc/nginx/ssl

Генерация самоподписанного сертификата

# Генерируем приватный ключ и самоподписанный сертификат одной командой
# -x509   — создать самоподписанный сертификат (не CSR)
# -nodes  — не шифровать ключ паролем (nginx читает без пароля)
# -days   — срок действия 365 дней
# -newkey rsa:2048 — новый RSA-ключ 2048 бит
# -keyout — путь к файлу ключа
# -out    — путь к файлу сертификата
openssl req -x509 -nodes -days 365 -newkey rsa:2048 \
  -keyout /etc/nginx/ssl/nextcloud.key \
  -out /etc/nginx/ssl/nextcloud.crt \
  -subj "/C=RU/ST=Moscow/L=Moscow/O=sirius-exam/CN=br-srv.sirius-exam.org"
# Устанавливаем правильные права на ключ (только root читает)
chmod 600 /etc/nginx/ssl/nextcloud.key

# Проверяем созданные файлы
ls -la /etc/nginx/ssl/
# Должны быть: nextcloud.crt и nextcloud.key

# Просмотр информации о сертификате
openssl x509 -in /etc/nginx/ssl/nextcloud.crt -text -noout | head -20
1.3
Настройка Nginx как обратного прокси
BR-SRV — HTTPS → Nextcloud:8080, заголовки X-Real-IP, HSTS
📦 Пакет nginx уже предустановлен.

Создание конфигурации виртуального хоста

Создаём отдельный файл конфигурации для Nextcloud (хорошая практика — не редактировать основной nginx.conf):

# Создаём директорию sites-available (если нет)
mkdir -p /etc/nginx/sites-available
mkdir -p /etc/nginx/sites-enabled

# Создаём конфиг для Nextcloud
nano /etc/nginx/sites-available/nextcloud.conf

Содержимое файла конфигурации:

server {
    listen 80;
    server_name _;

    # Перенаправляем HTTP → HTTPS
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name _;

    # Пути к SSL-сертификату и ключу
    ssl_certificate     /etc/nginx/ssl/nextcloud.crt;
    ssl_certificate_key /etc/nginx/ssl/nextcloud.key;

    # Современные настройки SSL
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;

    # HSTS — сообщаем браузеру использовать только HTTPS
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains; preload" always;

    # Основные заголовки безопасности
    add_header X-Content-Type-Options nosniff always;

    location / {
        # Проксируем запросы на Nextcloud (порт 8080 на localhost)
        proxy_pass http://127.0.0.1:8080;

        # Передаём реальный IP клиента в контейнер
        proxy_set_header X-Real-IP $remote_addr;

        # Передаём информацию о протоколе (https)
        proxy_set_header X-Forwarded-Proto $scheme;

        # Стандартные заголовки прокси
        proxy_set_header Host $host;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

        # Таймауты для медленных операций Nextcloud
        proxy_connect_timeout 300;
        proxy_send_timeout 300;
        proxy_read_timeout 300;

        # Размер буфера для загрузки файлов
        client_max_body_size 512M;
    }
}

Подключение конфига через символическую ссылку

# Создаём символическую ссылку из sites-enabled → sites-available
# Это стандартная практика: конфиг хранится в sites-available,
# а активируется через ссылку в sites-enabled
ln -s /etc/nginx/sites-available/nextcloud.conf /etc/nginx/sites-enabled/nextcloud.conf

# Проверяем что ссылка создана
ls -la /etc/nginx/sites-enabled/

Подключение sites-enabled к главному конфигу nginx

Нужно убедиться, что основной nginx.conf включает файлы из sites-enabled:

nano /etc/nginx/nginx.conf

Внутри блока http { } добавьте строку (если её нет):

# В блоке http { } добавляем include:
include /etc/nginx/sites-enabled/*.conf;

Проверка и запуск nginx

# Проверяем синтаксис всей конфигурации
nginx -t
# Ожидаем: syntax is ok / test is successful

# Запускаем nginx и добавляем в автозагрузку
systemctl enable --now nginx

# Или перезапускаем если уже запущен
systemctl restart nginx

# Проверяем статус
systemctl status nginx

Открытие портов в firewall

firewall-cmd --list-all

Создание каталогов и настройка прав
HQ-CLI — /home/Folder/work_shared (770) и /home/Folder/job_readonly (750)
Задание /home/Folder/ — родительский каталог.
/home/Folder/work_shared — права 770, владелец root, группа Work.
/home/Folder/job_readonly — права 750, владелец root, группа Job.

Создание структуры каталогов

# Создаём родительский каталог
mkdir -p /home/Folder

# Создаём подкаталог для группы Work
mkdir -p /home/Folder/work_shared

# Создаём подкаталог для группы Job
mkdir -p /home/Folder/job_readonly

Настройка владельца и прав

# work_shared: владелец root, группа Work, права 770
# 770 = rwxrwx--- (root:rw+x, Work:rw+x, other:нет доступа)
chown root:Work /home/Folder/work_shared
chmod 770 /home/Folder/work_shared

# job_readonly: владелец root, группа Job, права 750
# 750 = rwxr-x--- (root:rw+x, Job:r+x, other:нет доступа)
chown root:Job /home/Folder/job_readonly
chmod 750 /home/Folder/job_readonly

Проверка прав

# Проверяем права на каталоги
ls -la /home/Folder/
# Ожидаемый вывод:
# drwxrwx--- root Work  work_shared
# drwxr-x--- root Job   job_readonly

Расшифровка прав доступа:

КаталогПраваСимвольныеСмысл
work_shared770rwxrwx--- root и группа Work — полный доступ (чтение, запись, выполнение). Все остальные — нет доступа.
job_readonly750rwxr-x--- root — полный доступ. Группа Job — только чтение и переход в каталог (без записи). Все остальные — нет доступа.

Проверка Nextcloud через браузер

3.4
Проверка заголовка HSTS через curl
HQ-CLI — Strict-Transport-Security в ответе сервера
Задание Убедиться в наличии заголовка: Strict-Transport-Security: max-age=31536000; includeSubDomains; preload 
# С HQ-CLI выполняем curl — смотрим только заголовки ответа
# -k  — игнорируем ошибку самоподписанного сертификата
# -I  — выводим только заголовки (HEAD-запрос)
curl -k -I https://192.168.255.2

# Ожидаемый вывод содержит строку:
# Strict-Transport-Security: max-age=31536000; includeSubDomains; preload
# Можно отфильтровать только строку HSTS:
curl -k -I https://192.168.255.2 2>/dev/null | grep -i strict-transport
# Ожидаем:
# Strict-Transport-Security: max-age=31536000; includeSubDomains; preload
📌 Что означает HSTS max-age=31536000 — браузер запомнит что сайт использует только HTTPS на 1 год (31536000 секунд).
includeSubDomains — правило распространяется на все поддомены.
preload — разрешение на добавление в список предзагрузки браузеров.

Если заголовок не появляется — проверьте наличие add_header Strict-Transport-Security в конфиге nginx и перезапустите nginx.

  • В логах nginx на BR-SRV виден IP HQ-CLI
  • curl выводит заголовок Strict-Transport-Security
⚠
Возможные ошибки и решения
Диагностика проблем вариативной части
❌ docker-compose up: Cannot connect to the Docker daemon
Причина: служба Docker не запущена.
systemctl start docker
systemctl status docker
❌ nextcloud_app перезапускается, в логах: "SQLSTATE[HY000] Can't connect to MySQL"
Причина: контейнер db ещё не готов принимать соединения. Nextcloud стартовал раньше MySQL.
# Ждём 30–60 секунд и перезапускаем только app
docker restart nextcloud_app

# Или полностью пересоздаём стек
cd /opt/nextcloud
docker-compose down
docker-compose up -d
❌ nginx -t: "no such file or directory" для sites-enabled
Причина: директория sites-enabled не создана, или include не добавлен в nginx.conf.
# Создаём директорию
mkdir -p /etc/nginx/sites-available /etc/nginx/sites-enabled

# Проверяем что include присутствует в nginx.conf
grep "sites-enabled" /etc/nginx/nginx.conf

# Если нет — добавляем внутри блока http { }
# include /etc/nginx/sites-enabled/*.conf;
❌ curl: SSL certificate problem: self-signed certificate
Причина: забыли флаг -k для игнорирования самоподписанного сертификата.
# Всегда используйте -k с самоподписанным сертификатом
curl -k -I https://192.168.255.2
❌ Заголовок Strict-Transport-Security не появляется в curl
Причина: директива add_header отсутствует или nginx не перезапущен.
# Проверяем конфиг
grep -i hsts /etc/nginx/sites-available/nextcloud.conf
grep -i strict /etc/nginx/sites-available/nextcloud.conf

# Перезапускаем nginx
systemctl restart nginx

# Проверяем снова
curl -k -I https://192.168.255.2 | grep -i strict
❌ Permission denied при попытке войти в каталог от User1
Причина: пользователь не добавлен в группу Work, или права на каталог не установлены.
# Проверяем группы пользователя
id User1

# Если группы Work нет — добавляем
usermod -aG Work User1

# Пользователь должен перелогиниться чтобы группа применилась
su - User1

# Проверяем права каталога
ls -la /home/Folder/
❌ Nextcloud показывает страницу установки, хотя переменные заданы
Причина: контейнер был создан до появления переменных окружения, или volume уже содержит старую конфигурацию.
# Останавливаем и удаляем контейнеры вместе с данными
cd /opt/nextcloud
docker-compose down -v

# Очищаем volume-данные
rm -rf /opt/nextcloud/nextcloud_data/*
rm -rf /opt/nextcloud/mysql_data/*

# Пересоздаём
docker-compose up -d
❌ 502 Bad Gateway в nginx при обращении к Nextcloud
Причина: контейнер nextcloud_app не запущен или ещё инициализируется.
# Проверяем статус контейнеров
docker ps

# Проверяем что порт 8080 слушается
ss -tlnp | grep 8080

# Смотрим логи nextcloud
docker logs nextcloud_app --tail 30
